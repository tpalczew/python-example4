# try K=1 through K=35 and record testing accuracy
k_range = list(range(1, 36))
scores = []
for k in k_range:
    knn2 = KNeighborsClassifier(n_neighbors=k)
    knn2.fit(X, y)
    y_pred = knn2.predict(X)
    scores.append(metrics.accuracy_score(y, y_pred))
